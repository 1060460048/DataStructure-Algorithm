11 int cut_rod(const int *p, const int &list_len, const int &n)
12 {
13  if (n == 0)
14  return 0;
15  int q = INT_MIN;
16  // 问题规模降低时的最大收益
17  for (int i = 1; i <= min(list_len, n); i++)
18  {
19  if (q < p[i - 1] + cut_rod(p, list_len, n - i))
20  q = p[i - 1] + cut_rod(p, list_len, n - i);
21  }
22  return q;
23 }
