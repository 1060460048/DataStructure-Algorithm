97 // 优先级搜索
98 template <typename Tv, typename Te>
99 template <typename PU>
100 void Graph<Tv, Te>::pfs(int s, PU prioUpdater)
101 {
102  reset();
103  priority(s) = 0;
104  parent(s) = -1;
105  status(s) = VISITED; //将s加入PFS树中
106  // 将n-1个顶点和边加入PFS树中
107  while (1)
108  {
109  // 更新s所有邻居的优先级,并更新其父节点
110  for (int i = firstNbr(s); - 1 < i; i = nextNbr(s, i))
111  {
112  prioUpdater(this, s, i);
113  }
114 
115  // 在所有未加入PFS树的节点中找出优先级最高的点,选为下一个
116  for (int shortest = INT_MAX, w = 0; w < numVertex(); w++)
117  {
118  if (status(w) == UNDISCOVERED)
119  if (priority(w) < shortest)
120  {
121  shortest = priority(w);
122  s = w;
123  }
124  }
125  if (status(s) == VISITED)
126  break;
127  status(s) = VISITED; //将新的s加入PFS树
128  status(parent(s), s) = TREE;
129  }
130 
131  // 标记生成树的边
132  for (int i = 0; i < numVertex(); i++)
133  if (!(parent(i) < 0))
134  status(parent(i), i) = TREE;
135 
136  return;
137 }
